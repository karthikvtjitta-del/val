        // Advanced floating hearts with mouse follow (inspired by pro animations)
        const canvas = document.getElementById('particles');
        const ctx = canvas.getContext('2d');
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
        let hearts = [];
        let mouse = { x: 0, y: 0 };

        class Heart {
            constructor() {
                this.x = Math.random() * canvas.width;
                this.y = canvas.height + 100;
                this.size = Math.random() * 20 + 10;
                this.speedX = Math.random() * 2 - 1;
                this.speedY = Math.random() * -3 - 1;
                this.opacity = 0;
                this.vx = 0;
                this.vy = 0;
            }
            update() {
                const dx = mouse.x - this.x;
                const dy = mouse.y - this.y;
                const dist = Math.sqrt(dx*dx + dy*dy);
                if (dist < 100) {
                    this.vx += dx * 0.01;
                    this.vy += dy * 0.01;
                }
                this.x += this.speedX + this.vx * 0.2;
                this.y += this.speedY + this.vy * 0.2;
                this.opacity += 0.02;
                this.vx *= 0.95;
                this.vy *= 0.95;
                if (this.y < 0) this.y = canvas.height + 100;
            }
            draw() {
                ctx.save();
                ctx.globalAlpha = this.opacity;
                ctx.fillStyle = '#ff69b4';
                ctx.font = `${this.size}px Arial`;
                ctx.fillText('üíñ', this.x, this.y);
                ctx.restore();
            }
        }

        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            if (hearts.length < 50) hearts.push(new Heart());
            hearts.forEach((heart, i) => {
                heart.update();
                heart.draw();
                if (heart.opacity > 1) hearts.splice(i, 1);
            });
            requestAnimationFrame(animate);
        }
        animate();

        document.addEventListener('mousemove', e => {
            mouse.x = e.clientX;
            mouse.y = e.clientY;
